package DSA.Stack;
// stack sytanx.
// first top, arr, capacity 

class Stack{
    int top;
    int[] arr; 
    int capacity;
    Stack(int size){
        capacity = size;
        arr = new int[size];
        top = -1; 
    }
}

// push operation
void push(int value){
    if(top == capacity -1){
        System.out.println("Stack is full");
        return;
    }
    arr[++top] = value;
    System.out.println(value + " pushed ");
}
// pop  operation
int pop(){
    if(top == -1){
        System.out.println("Stack Underflow");
        return -1;
    }
    return arr[top--];
}
// peak operation
int peek(){
    if(top == -1){
        System.out.println("Stack is Empty");
        return -1;
    }
    return arr[top];
}
//check is Empty
boolean isEmpty(){
    return top == -1;
}
}
